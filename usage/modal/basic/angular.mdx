```html
<li-modal
  heading="Today is your lucky day"
  id="modal">
  <p>Hello. Welcome. What a pleasure it is to have you.</p>
  <li-button
    slot="action"
    variant="secondary"
    onclick="closeModal()">
    Cancel
  </li-button>
  <li-button slot="action"> Primary Action </li-button>
</li-modal>

<li-button onclick="openModal()"> Open Modal </li-button>

<script>
  const modal = document.getElementById('modal');
  function openModal() {
    modal.opened = true;
  }
  function closeModal() {
    modal.opened = false;
  }
</script>
```
